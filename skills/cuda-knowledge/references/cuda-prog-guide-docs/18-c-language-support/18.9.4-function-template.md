### 18.9.4. Function Template’āü


    template <typename T>
    __device__ bool func(T x)
    {
       ...
       return (...);
    }
    
    template <>
    __device__ bool func<int>(T x) // Specialization
    {
       return true;
    }
    
    // Explicit argument specification
    bool result = func<double>(0.5);
    
    // Implicit argument deduction
    int x = 1;
    bool result = func(x);
