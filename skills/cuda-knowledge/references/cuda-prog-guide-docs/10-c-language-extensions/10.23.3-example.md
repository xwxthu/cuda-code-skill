### 10.23.3. Example’āü


The following code implements a mutex with exponential back-off.


    __device__ void mutex_lock(unsigned int *mutex) {
        unsigned int ns = 8;
        while (atomicCAS(mutex, 0, 1) == 1) {
            __nanosleep(ns);
            if (ns < 256) {
                ns *= 2;
            }
        }
    }
    
    __device__ void mutex_unlock(unsigned int *mutex) {
        atomicExch(mutex, 0);
    }
